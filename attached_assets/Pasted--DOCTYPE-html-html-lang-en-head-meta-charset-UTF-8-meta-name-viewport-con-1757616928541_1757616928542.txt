    <script>
        // Create chaos particles for the "before" state
        const particlesContainer = document.getElementById('chaosParticles');
        for (let i = 0; i < 20; i++) {
            const particle = document.createElement('div');
            particle.className = 'particle';
            particle.style.left = Math.random() * 100 + '%';
            particle.style.top = Math.random() * 100 + '%';
            particle.style.animationDelay = Math.random() * 3 + 's';
            particle.style.animation = `float ${3 + Math.random() * 2}s infinite`;
            particlesContainer.appendChild(particle);
        }

        // Add floating animation
        const style = document.createElement('style');
        style.textContent = `
            @keyframes float {
                0%, 100% {
                    transform: translate(0, 0);
                    opacity: 0.6;
                }
                25% {
                    transform: translate(10px, -10px);
                    opacity: 0.3;
                }
                50% {
                    transform: translate(-10px, 10px);
                    opacity: 0.6;
                }
                75% {
                    transform: translate(5px, -5px);
                    opacity: 0.3;
                }
            }
        `;
        document.head.appendChild(style);

        // Animate metrics on load
        setTimeout(() => {
            document.querySelectorAll('.after-state .metric-fill').forEach(fill => {
                fill.style.width = fill.style.getPropertyValue('--after-width');
            });
        }, 500);

        // Animate chart lines
        document.querySelectorAll('.chart-line').forEach(line => {
            const length = line.getTotalLength();
            line.style.strokeDasharray = length;
            line.style.strokeDashoffset = length;
            line.style.animation = `drawLine 2s forwards`;
        });

        const chartStyle = document.createElement('style');
        chartStyle.textContent = `
            @keyframes drawLine {
                to {
                    stroke-dashoffset: 0;
                }
            }
        `;
        document.head.appendChild(chartStyle);
    </script>
